create table Users (
	id int primary key identity (1,1),
	username nvarchar(255) unique,
	[password] nvarchar(255),
	email nvarchar(255),
	bio nvarchar(MAX),
	joined date default getdate(),
	picture nvarchar(255) default 'anonymous.png'
)

create table Genre (
	id int primary key identity (1,1),
	name nvarchar(255)
)

create table Games (
	id int primary key identity (1,1),
	genre_Id int foreign key references Genre(id),
	title nvarchar(255),
	blurb nvarchar(MAX),
	release date,
	developer nvarchar(255),
	publisher nvarchar(255),
	image nvarchar(255) default 'noImage.png'
)

create table Reviews (
	id int primary key identity (1,1),
	game_Id int foreign key references Games(id),
	appUser_Id int foreign key references Users(id),
	title nvarchar(255),
	review nvarchar(MAX),
	gameplay int,
	presentation int,
	engagement int,
	difficulty int,
	replayable int,
	created date default getdate()
)
